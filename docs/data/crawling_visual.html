Playwright 구현 패턴 
from playwright.async_api import async_playwright
from bs4 import BeautifulSoup
import re

async def crawl_reviews(goods_id: str):
    async with async_playwright() as p:
        browser = await p.chromium.launch(headless=True)
        context = await browser.new_context()
        page = await context.new_page()

        # 세션 초기화 (쿠키 발급)
        await page.goto("https://www.aboutpet.co.kr/shop/home/", wait_until="networkidle")

        # 후기 직접 호출 — JS 트리거 불필요 (goodsCstrtTpCd=ITEM 필수)
        resp = await page.request.post(
            "https://www.aboutpet.co.kr/goods/getGoodsEntireCommentList",
            form={
                "goodsId": goods_id,
                "goodsCstrtTpCd": "ITEM",  # 필수 — 누락 시 totalCount=0
                "page": "1",
                "sidx": "", "sord": "",
                "optGoodsId": "", "detailYn": "Y",
                "petKindNm": "", "petAge": "", "petWeight": "",
                "pageGoodsEstmNo": "",
            }
        )
        html = await resp.text()
        soup = BeautifulSoup(html, "html.parser")
        total_pages = int(re.search(r"totalPageCount\s*=\s*(\d+)", html).group(1))

        reviews = []
        for box in soup.select('div[name="estmDataArea"]'):
            estm_no  = box.get("data-goods-estm-no")
            nickname = box.select_one(".writer-info .ids")
            date     = box.select_one(".writer-info .date")
            stars_el = box.select_one(".stars.sm")

            # 별점: class p_X_Y → float (p_5_0=5.0, p_4_5=4.5)
            star_cls = next((c for c in (stars_el.get("class", []) if stars_el else [])
                             if c.startswith("p_")), None)
            rating = float(star_cls[2:].replace("_", ".")) if star_cls else None

            # 구매 유형 (항상 존재)
            pl = box.select_one(".purchase-label")
            purchase_label = ("first" if pl and "first" in pl.get("class", [])
                              else "repeat" if pl else None)

            # 후기 본문
            msg = box.select_one(".msgs")

            # 펫 프로필 (조건부) — div.spec
            spec = box.select_one("div.spec")
            pet = {}
            if spec:
                ems = spec.select("em")
                gender_el = spec.select_one("em.b > i.g")
                name_el   = spec.select_one("em.b")
                pet = {
                    "name":   name_el.get_text(strip=True).split("(")[0] if name_el else None,
                    "gender": gender_el.get_text(strip=True) if gender_el else None,
                    "age":    ems[1].get_text(strip=True) if len(ems) > 1 else None,
                    "weight": ems[2].get_text(strip=True) if len(ems) > 2 else None,
                    "breed":  ems[3].get_text(strip=True) if len(ems) > 3 else None,
                }

            # review_info (카테고리 의존 — 식품류 없음)
            review_info = {
                li.select_one(".dt").get_text(strip=True): li.select_one(".dd").get_text(strip=True)
                for li in box.select("ul.satis li")
                if li.select_one(".dt") and li.select_one(".dd")
            }

            reviews.append({
                "review_no": estm_no,
                "nickname": nickname.get_text(strip=True) if nickname else None,
                "date": date.get_text(strip=True) if date else None,
                "rating": rating,
                "purchase_label": purchase_label,
                "review_text": msg.get_text(strip=True) if msg else "",
                "pet_profile": pet or None,
                "review_info": review_info,  # {} if food category
            })

        await browser.close()
        return reviews
ℹ️ goodsCstrtTpCd=ITEM 누락 시 API가 빈 응답 반환. 세션 쿠키 유지를 위해 반드시 메인 페이지 방문 후 API 호출.

주의 사항
⚠️ 평점 스케일 불일치: 상품목록 API(data-goodsstarsavgcnt) = 10점 만점, 리뷰 별점(p_X_Y class) = 5점 만점. Silver 레이어 정제 시 5점 기준으로 통일.
